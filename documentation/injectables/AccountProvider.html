import { HttpClient, HttpHeaders } from '@angular/common/http';
import { Injectable } from '@angular/core';

//RxJS
import { Observable } from 'rxjs/Observable';

import { ConfigProvider } from '../config/config';
@Injectable()
export class AccountProvider {
  private LOGIN_URL:string = 'login';
  private SIGNUP_URL:string = 'signup';
  private HTTP_HEADERS:HttpHeaders = null;

  constructor(public http: HttpClient, public configProv: ConfigProvider) {
    this.HTTP_HEADERS = new HttpHeaders({
      'Content-Type': 'application/json'
    });
  }

  public doLogin(values:any):Observable<Object> {
    return this.http.post(`${this.configProv.getApiServer()}${this.LOGIN_URL}`, values, { headers: this.HTTP_HEADERS });
  }

  public singUp(values:any):Observable<Object> {
    return this.http.post(`${this.configProv.getApiServer()}${this.SIGNUP_URL}`, values, { headers: this.HTTP_HEADERS });
  }
}
