import { Component } from '@angular/core';
import { IonicPage, NavController, NavParams, Modal, AlertController, ModalController, ModalOptions } from 'ionic-angular';
import { PaymentsPage } from './../payments/payments';

import { ModalProcessingDataPage } from './../modal-processing-data/modal-processing-data';

import { Camera, CameraOptions } from '@ionic-native/camera';

//@IonicPage()
@Component({
  selector: 'page-id-verification',
  templateUrl: 'id-verification.html',
})
export class IdVerificationPage {
  private picture:string = null;
  private pattern:RegExp = /^(\d{2})\/(d{2})$/;
  private cameraOptions: CameraOptions = {
    quality: 100,
    destinationType: this.camera.DestinationType.DATA_URL,
    encodingType: this.camera.EncodingType.JPEG,
    mediaType: this.camera.MediaType.PICTURE
  }

  constructor(public navCtrl: NavController, public navParams: NavParams, public alertCtrl: AlertController, public modalCtrl: ModalController, private camera: Camera) {
    console.log(this.pattern.test('12/22'))
    console.log(this.pattern.exec('12/22'));
  }

  private takePicture():void {
    this.camera.getPicture(this.cameraOptions).then(data => {
      this.picture = `data:image/jpeg;base64,${data}`;
      console.log(this.picture);
    }).catch(error => console.error(error));
  }

  private openPayment() {
    const myModalOptions: ModalOptions = {
      enableBackdropDismiss: false,
      showBackdrop: false,
      cssClass: 'my-modal-inner'
    };
    let modal = this.modalCtrl.create(ModalProcessingDataPage, myModalOptions);
    modal.present();
    setTimeout(() => {
      modal.dismiss();
      this.navCtrl.setRoot(PaymentsPage);
    }, 2000);
  }

}
