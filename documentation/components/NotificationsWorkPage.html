import { Component, ViewChild } from '@angular/core';
import { IonicPage, NavController, NavParams, Tabs, AlertController } from 'ionic-angular';

export interface Notification{
  img?:string;
  projectName?:string;
  userName?:string;
  companyPosition?:string;
  description?:string;
  note?:string;
}

@IonicPage()
@Component({
  selector: 'page-notifications-work',
  templateUrl: 'notifications-work.html',
})
export class NotificationsWorkPage {
  private notifications:Array<Notification> = [
    {
      img: '../../assets/imgs/foto_not.png',
      projectName: 'Pepsico Photoshot',
      userName: 'Enrique',
      companyPosition: 'Ceo Pepsi',
      description: 'View description',
      note: '5.0'
    },
    {
      img: '../../assets/imgs/foto_not.png',
      projectName: 'Sabritas campaña',
      userName: 'Ana',
      companyPosition: 'Producer',
      description: 'View description',
      note: '4.0'
    }
  ]
  /**
   * Contructor of the class
   * @param navCtrl   NavController
   * @param navParams NavParams
   * @param alertCtrl AlertController
   */
  constructor(public navCtrl: NavController, public navParams: NavParams, private alertCtrl: AlertController) {
  }

  /**
   * Delete the notification from list
   * @param i Index of notification to delete
   */
  private delete(i:number):void {
    let alert = this.alertCtrl.create({
      title: 'Are you sure to delete this notification?',
      buttons: [
        {
          role: 'cancel',
          text: 'Cancel'
        },
        {
          text: 'Ok',
          handler: () => {
            this.notifications.splice(i, 1);
          }
        }
      ]
    });
    alert.present();
  }

}
