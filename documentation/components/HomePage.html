import { Component } from '@angular/core';
import { NavController } from 'ionic-angular';

//Angular Forms
import { Validators, FormBuilder, FormGroup, FormControl } from '@angular/forms';

//interfaces
import { Sesion } from '../../model/interfaces/interfaces';

//providers
import { AccountProvider } from '../../providers/account/account';
import { ConfigProvider } from '../../providers/config/config';

//Pages
import { SelectRegisterPage } from '../select-register/select-register';

@Component({
  selector: 'page-home',
  templateUrl: 'home.html'
})
export class HomePage {
  /**
   * Form to login
   */
  private loginForm: FormGroup;

  constructor(public navCtrl: NavController, private formBuilder: FormBuilder, private accountProv: AccountProvider, private configProv: ConfigProvider) {
    this.loginForm = this.formBuilder.group({
      email: new FormControl('', Validators.compose([
        Validators.required, Validators.email
      ])),
      password: new FormControl('', Validators.required)
    });
  }

  ionViewDidLoad() {
    this.configProv.getSesionObject().then(sesionObject => {
      if (sesionObject != null) {
        this.navCtrl.setRoot(SelectRegisterPage);
      }
    }).catch(error => console.log(error));
  }

  /**
   * Do the login
   * @return Void
   */
  login(){
    if (this.loginForm.valid) {
      let values:Sesion = this.loginForm.getRawValue();
      this.configProv.setSesionObject(values);
      this.navCtrl.setRoot(SelectRegisterPage);
      /*this.accountProv.doLogin(values).subscribe(data => {
        console.log(data);
      }, error => console.error(error));*/
    }else {
      this.loginForm.controls.email.markAsTouched();
      this.loginForm.controls.password.markAsTouched();
    }
  }

  register(){
    this.navCtrl.push(SelectRegisterPage);
  }

}
